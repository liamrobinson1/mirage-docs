[docs]def get_spacetrack_client(
    is_async: bool = False,
) -> Union[SpaceTrackClient, AsyncSpaceTrackClient]:
    """Gets client for the Space-Track API

    :param is_async: Whether the produced client operates asynchronously, defaults to False
    :type is_async: bool, optional
    :return: Client used to execute Space-Track API requests
    :rtype: Union[SpaceTrackClient, AsyncSpaceTrackClient]
    """
    (user, pword) = (
        os.environ["SPACETRACK_USERNAME"],
        os.environ["SPACETRACK_PASSWORD"],
    )
    if is_async:
        return AsyncSpaceTrackClient(identity=user, password=pword)
    else:
        return SpaceTrackClient(identity=user, password=pword)

async def _download_latest_tles(
    preset: str = "all", epoch_tol_days: int = 30, format: str = "tle"
) -> None:
    """Downloads TLE catalog from Space-Track

    :param preset: Preset TLE set, either "all", "leo", or "geo", defaults to "all"
    :type preset: str, optional
    :param epoch_tol_days: Oldest TLE to consider, defaults to 30
    :type epoch_tol_days: int, optional
    :param format: Format for output, defaults to 'tle'
    :type format: str, optional
    """
    st = get_spacetrack_client(is_async=True)
    cat_path = _cat_path(preset)
    async with st:
        if preset.lower() == "geo":
            data = await st.tle_latest(
                iter_lines=True,
                ordinal=1,
                epoch=f">now-{epoch_tol_days}",
                mean_motion=op.inclusive_range(0.99, 1.01),
                eccentricity=op.less_than(0.01),
                format=format,
            )
        elif preset.lower() == "leo":
            data = await st.tle_latest(
                iter_lines=True,
                ordinal=1,
                epoch=f">now-{epoch_tol_days}",
                semimajor_axis=op.less_than(8378),
                format=format,
            )
        elif preset.lower() == "all":
            data = await st.tle_latest(
                iter_lines=True,
                ordinal=1,
                epoch=f">now-{epoch_tol_days}",
                format=format,
            )

        with open(cat_path, "w+") as fp:
            async for line in data:
                fp.write(line + "\n")



[docs]def download_catalog_tles(
    preset: str = "all",
    format: str = "tle",
    only_run_if_empty: bool = False,
) -> None:
    """Downloads catalog TLEs for a given preset

    :param preset: Preset TLE set, either "all", "leo", or "geo", defaults to "all"
    :type preset: str, optional
    :param format: Format for output, defaults to 'tle'
    :type format: str, optional
    :param only_run_if_empty: If True, checks whether a catalog of the given preset already exists. If it does, the download is skipped
    :type only_run_if_empty: bool
    """
    if only_run_if_empty and os.path.exists(_cat_path(preset)):
        return
    print(f"Downloading TLE catalog for preset '{preset}'")
    loop = asyncio.new_event_loop()
    loop.run_until_complete(
        _download_latest_tles(preset=preset, format=format)
    )
    print(f"Finished downloading TLE catalog for preset '{preset}'")

[docs]def get_tles_between(
    st: SpaceTrackClient,
    satnum: int,
    dtimes: np.ndarray[datetime.datetime],
) -> list[tuple[str, str]]:
    """Gets all TLEs published in the range of datetimes passed

    :param st: Spacetrack client to access the API with
    :type st: SpaceTrackClient
    :param satnum: Satellite number, ex. 25544 for ISS ZARYA
    :type satnum: int
    :param dtimes: Datetimes, assumed to be UTC, in chronologically ascending order
    :type dtimes: np.ndarray[datetime]
    :raises ValueError: If the object decays during the timespan
    :return: List of TLE lines 1 and 2
    :rtype: list[tuple[str, str]]
    """
    idtime, fdtime = dtimes[0], dtimes[-1]
    idtime = datetime.datetime(
        year=idtime.year, month=idtime.month, day=idtime.day
    )
    fdtime = datetime.datetime(
        year=fdtime.year, month=fdtime.month, day=fdtime.day
    )
    query = {
        "norad_cat_id": str(satnum),
        "orderby": "epoch asc",
        "epoch": op.inclusive_range(idtime, fdtime),
    }

    # Get the closest TLE to the specified date
    tles = st.tle(**query)
    for tle in tles:
        if tle["DECAYED"] == "1":
            raise ValueError(
                f"Object decayed sometime within {query['epoch']}"
            )
    return [(tle["TLE_LINE1"], tle["TLE_LINE2"]) for tle in tles]




[docs]def first_tle_after(
    st: SpaceTrackClient, satnum: int, dtime: datetime
) -> tuple[str, str]:
    """Gets the first TLE produced for a given NORAD SAT ID after the datetime provided

    :param st: Spacetrack client to access the API with
    :type st: SpaceTrackClient
    :param satnum: Satellite number, ex. 25544 for ISS ZARYA
    :type satnum: int
    :param dtime: Datetime to query at, assumed to be UTC
    :type dtime: datetime
    :raises ValueError: If the object has already decayed at the given epoch
    :return: Line 1 and 2 of the next TLE
    :rtype: tuple[str, str]
    """
    datestr = dtime.strftime("%Y-%m-%d")
    # Define the search parameters
    query = {
        "norad_cat_id": str(satnum),
        "orderby": "epoch asc",
        "limit": 1,
        "epoch": f">{datestr}",
    }

    # Get the closest TLE to the specified date
    closest_tle = st.tle(**query)[0]
    if closest_tle["DECAYED"] == "1":
        raise ValueError(f"Object decayed sometime before {datestr}")
    return closest_tle["TLE_LINE1"], closest_tle["TLE_LINE2"]

        flatten = (

            .replace(" ", "")

        )[0]
        raise NameError(
            f"No satellite found with name {repr(name)}, did you mean {repr(closest)}?"
        )

[docs]def build_satdef_array() -> SatDefArray:
    download_catalog_tles(
        format="3le"
    )  # Make sure we have a catalog to work with
    satdefs = SatDefArray()
    with open(
        os.path.join(os.environ["TLECATDIR"], "tle_latest_all.cat"), "r"
    ) as f:
        for line in f.readlines():
            line = line.replace("\n", "")
            if line[0] == "0":
                name = line[2:]
            if line[0] == "1":
                satnum = int(line[1:7].replace(" ", ""))
                ides = line[9:17].replace(" ", "")
                satdefs.append(SatDef(name, satnum, ides))
    return satdefs
