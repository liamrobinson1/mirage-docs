def _finditem(obj, key):
    if key in obj:
        return obj[key]
    for k, v in obj.items():
        if isinstance(v, dict):
            item = _finditem(v, key)
            if item is not None:
                return item


def _recursive_items(dictionary):
    for key, value in dictionary.items():
        if type(value) is dict:
            yield from _recursive_items(value)
        else:
            yield (key, value)

[docs]def download_model_file(model_file: str):
    """Downloads a model file (.obj) from the light curve models repository

    :param model_file: Model file name (not path) ending in .obj
    :type model_file: str
    """
    assert (
        model_file[-4:] == ".obj"
    ), "Model file must be of the form *.obj"
    download_listing()
    print(
        f"Attempting to download {model_file} and its associated material file from the model repository..."
    )
    mtl_file = f"{model_file[:-4]}.mtl"
    get_files = [model_file, mtl_file]
    for file in get_files:
        path = get_model_path(file)
        url = f"{_BASE_URL}/{path}"
        req = requests.get(url)
        if req.status_code == 200:
            with open(
                os.path.join(os.environ["MODELDIR"], file), "w"
            ) as f:
                f.write(req.text)
        else:
            raise FileNotFoundError(
                f"{file} not found in the repository! Aborting."
            )

    print(f"Model files were downloaded successfully!")
