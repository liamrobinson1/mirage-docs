[docs]def render_video(
    pre_render_fcn: Callable,
    render_fcn: Callable,
    post_render_fcn: Callable,
    frame_count: int,
    fname: str = None,
    framerate: int = 60,
    quality: int = 9,
    background_color: str = "white",
) -> None:
    """Renders a video given callables each stage of the recording

    :param pre_render_fcn: Called once before the first frame
    :type pre_render_fcn: Callable(pl: pv.Plotter)
    :param render_fcn: Called each frame
    :type render_fcn: Callable(pl: pv.Plotter, i: int)
    :param post_render_fcn: Called after each frame, before the next (only needed for niche effects)
    :type post_render_fcn: Callable(pl: pv.Plotter, i: int)
    :param frame_count: Number of frames to render
    :type frame_count: int
    :param fname: File name to save as, defaults to None
    :type fname: str, optional
    :param framerate: Frame rate of movie [frames/second], defaults to 60
    :type framerate: int, optional
    :param quality: Quality of movie file, defaults to 9, less than 10
    :type quality: int, optional
    :param background_color: Background color of plot window, defaults to "white"
    :type background_color: str, optional
    """
    pl = pv.Plotter()
    if fname is not None:
        if fname[-3:] == "mp4":
            pl.open_movie(fname, framerate=framerate, quality=quality)
        elif fname[-3:] == "gif":
            pl.open_gif(fname, fps=framerate)
    pl.set_background(background_color)
    pre_render_fcn(pl)
    pl._on_first_render_request()
    pl.render()
    for i in range(frame_count):
        render_fcn(pl, i)
        if fname is not None:
            pl.write_frame()
        else:
            pl.update()
        post_render_fcn(pl, i)
    pl.close()




[docs]def vis_attitude_motion(
    obj: SpaceObject,
    quat: np.ndarray,
    fname: str = None,
    framerate: int = 60,
    quality: int = 9,
    background_color: str = "white",
) -> None:
    """Visualizes attitude motion in terms of quaternions, applied to a 3D model

    :param obj: Object to visualize orientations with
    :type obj: SpaceObject
    :param quat: Orientation history
    :type quat: np.ndarray [nx4]
    :param fname: File name to save movie to, defaults to `None`. If `None`, no video saved
    :type fname: str, optional
    :param framerate: Frame rate of movie [frames/second], defaults to 60
    :type framerate: int, optional
    :param quality: Quality of movie file, defaults to 9, less than 10
    :type quality: int, optional
    :param background_color: Background color of plot window, defaults to "white"
    :type background_color: str, optional
    """
    pl = pv.Plotter()
    if fname is not None:
        if fname[-3:] == "mp4":
            pl.open_movie(fname, framerate=framerate, quality=quality)
        elif fname[-3:] == "gif":
            pl.open_gif(fname, fps=framerate)
    pl.set_background(background_color)
    pl.add_mesh(obj._mesh)
    o_obj = obj._mesh.copy()
    pl._on_first_render_request()
    pl.render()
    for i in range(quat.shape[0]):
        obj.rotate_by_quat(quat[i, :])
        if fname is not None:
            pl.write_frame()
        else:
            pl.update()
        obj._mesh.copy_from(o_obj)
    pl.close()

def _build_topo_interpolator() -> Callable:
    """Builds interpolators for elevation and bathymetry data from internal high-res image datasets

    :return: Interpolator that takes in latitude [rad] and longitude [rad] and returns elevation in km
    :rtype: Callable
    """
    max_elev_meters, min_bathy_meters = 6400, -8000
    with Image.open(
        os.path.join(_TEX_DIR, "earth_elev21600x10800.png")
    ) as im:
        im_elev = np.array(im, dtype=np.uint8)
        _LAT_SPACES = np.linspace(
            np.pi / 2, -np.pi / 2, im_elev.shape[0]
        )
        _LON_SPACES = np.linspace(-np.pi, np.pi, im_elev.shape[1])
        elev_interp = lambda lat, lon: scipy.interpolate.interpn(
            points=(
                _LAT_SPACES,
                _LON_SPACES,
            ),
            values=im_elev
            / 255.0
            * max_elev_meters,  # Scales image values to
            xi=(lat, lon),
            method="linear",
        )
    with Image.open(
        os.path.join(_TEX_DIR, "gebco_08_rev_bath_21600x10800.png")
    ) as im:
        im_bathy = np.array(im, dtype=np.uint8)
        _LAT_SPACES = np.linspace(
            np.pi / 2, -np.pi / 2, im_bathy.shape[0]
        )
        _LON_SPACES = np.linspace(-np.pi, np.pi, im_bathy.shape[1])
        bathy_interp = lambda lat, lon: scipy.interpolate.interpn(
            points=(
                _LAT_SPACES,
                _LON_SPACES,
            ),
            values=(1 - im_bathy / 255.0)
            * min_bathy_meters,  # Scales image values to
            xi=(lat, lon),
            method="linear",
        )

    return (
        lambda lat, lon: bathy_interp(lat, lon) / 1e3
        + elev_interp(lat, lon) / 1e3
    )  # /1e3 to output meters

[docs]def plot_earth(
    pl: pv.Plotter,
    mode: str = "eci",
    date: datetime.datetime = None,
    high_def: bool = False,
    stars: bool = False,
    night_lights: bool = False,
    elevation: bool = False,
    use_elevation_texture: bool = False,
    atmosphere: bool = False,
    lighting: bool = True,
) -> pv.PolyData:
    """Plots the Earth as a sphere in [km]

    :param pl: Plotter to add the Earth to
    :type pl: pv.Plotter
    :return: Generated spherical mesh
    :rtype: pv.PolyData
    """
    pl.enable_terrain_style(True, True)
    if elevation:
        interp_elev = _build_topo_interpolator()
    global earth_night_lights, earth_atmosphere, atmos_ecef
    sunset_tol, res = np.deg2rad(1), 500
    assert mode in ["eci", "ecef"], "Mode must be 'eci' or 'ecef'"
    (_, _, r3) = axis_rotation_matrices()
    rotm = r3(np.pi / 2)
    if mode == "eci":
        rotm = rotm @ ecef_to_eci(date)
    rotm4 = np.vstack(
        (np.hstack((rotm, np.zeros((3, 1)))), [[0, 0, 0, 1]])
    )
    rotm4_nl = rotm4 @ np.vstack(
        (np.hstack((r3(-np.pi / 2), np.zeros((3, 1)))), [[0, 0, 0, 1]])
    )

    if "earth" not in pl.actors:
        if date is not None and mode == "eci" and lighting:
            pl.remove_all_lights(only_active=True)
            light = pv.Light(
                intensity=1.5,
                # position=40e3 * hat(sun(date_to_jd(date)))[0, :],
            )
            sun_ra_dec = eci_to_ra_dec(sun(date_to_jd(date)))
            light.set_direction_angle(
                np.rad2deg(sun_ra_dec[1][0]),
                np.rad2deg(sun_ra_dec[0][0]),
            )
            pl.add_light(light, only_active=True)

        lat_space = np.linspace(-np.pi / 2, np.pi / 2, 50)
        lon_space = np.linspace(-np.pi, np.pi, 100)
        alt_space = np.linspace(0, 100, 3)
        latg, long, altg = np.meshgrid(
            lat_space, lon_space, alt_space, indexing="ij"
        )
        atmos_ecef = lla_to_itrf(
            latg.flatten(), long.flatten(), altg.flatten()
        )
        ecefg = atmos_ecef.reshape((*latg.shape, 3))
        earth_atmosphere = pv.StructuredGrid(
            ecefg[:, :, :, 0], ecefg[:, :, :, 1], ecefg[:, :, :, 2]
        )
        earth_atmosphere["density"] = 0 * altg.flatten()
        atmos_ecef = np.vstack(
            (
                ecefg[:, :, :, 0].flatten(order="f"),
                ecefg[:, :, :, 1].flatten(order="f"),
                ecefg[:, :, :, 2].flatten(order="f"),
            )
        ).T

        if atmosphere:
            # atmos_ecef = atmos_ecef @ r3(np.pi / 2).T
            ang_atmos = -sun_angle_to_horizon(
                date, atmos_ecef
            ).flatten()
            is_day = ang_atmos < -sunset_tol
            earth_atmosphere["density"][~is_day] = 1

            pl.add_volume(
                earth_atmosphere,
                cmap="gray",
                opacity=[0, 0.1],
                opacity_unit_distance=200,
                name="atmosphere",
                show_scalar_bar=False,
                culling="back",
            )

        a = (
            AstroConstants.earth_r_eq
        )  # Semi-major axis of Earth ellipsoid
        b = a * (1 - AstroConstants.earth_f)  # Semi-minor axis
        earth = pv.ParametricEllipsoid(
            a,
            a,
            b,
            u_res=res,
            v_res=res,
            w_res=res,
            min_u=-np.pi / 2 + 0.00001,
            max_u=3 * np.pi / 2 - 0.00001,
        )
        sph_hat_pts = hat(earth.points)
        earth.active_t_coords = np.zeros((earth.points.shape[0], 2))
        earth.active_t_coords[:, 0] = 0.5 + np.arctan2(
            -sph_hat_pts[:, 0], sph_hat_pts[:, 1]
        ) / (2 * np.pi)
        earth.active_t_coords[:, 1] = (
            0.5 + np.arcsin(sph_hat_pts[:, 2]) / np.pi
        )

        if elevation:
            (lat_pts, lon_pts, _) = itrf_to_lla(earth.points)
            lon_shifted = wrap_to_pi(lon_pts + np.pi / 2) * 0.99999
            elev_at_pts = interp_elev(lat_pts, lon_shifted)
            elev_at_pts[0] = np.abs(np.min(elev_at_pts))
            earth["elev_vector"] = (
                20 * elev_at_pts.reshape((-1, 1)) * hat(earth.points)
            )
            earth["elev_scalar"] = elev_at_pts
            earth.warp_by_vector("elev_vector", inplace=True)

        if high_def:
            tex_path = os.path.join(
                _TEX_DIR,
                "blue_marble.jpg",
            )
        else:
            tex_path = os.path.join(
                _TEX_DIR,
                "earth_tex.jpg",
            )
        etex = pv.Texture(tex_path)
        pl.add_mesh(
            earth,
            texture=etex if not use_elevation_texture else None,
            cmap="topo",
            smooth_shading=True,
            name="earth",
            scalars="elev_scalar" if use_elevation_texture else None,
        )
        pl.set_background("black")  # Because obviously, space and stuff
        if stars:
            cubemap = pv.cubemap(
                os.path.join(_TEX_DIR, "16k"), ext=".png"
            )
            pl.add_actor(cubemap.to_skybox())

        if night_lights:
            pts = np.fromfile(
                os.path.join(_TEX_DIR, "night_lights_ecef_tol_255.pkl")
            ).reshape((-1, 4))
            if date is not None:
                ang_nl = -sun_angle_to_horizon(
                    date, pts[:, :3]
                ).flatten()
                ang_atmos = -sun_angle_to_horizon(
                    date, atmos_ecef
                ).flatten()
                pts[:, 3] = (
                    np.clip(ang_nl, -sunset_tol, sunset_tol)
                    / sunset_tol
                    + 1
                )

            earth_night_lights = pv.PolyData(pts[:, :3])
            pl.add_points(
                earth_night_lights,
                style="points_gaussian",
                color="#ffffe6",
                point_size=0.5,
                opacity=pts[:, 3],
                name="earth_night_lights",
            )

    pl.actors["earth"].user_matrix = rotm4
    if night_lights:
        if date is not None:
            ang_nl = -sun_angle_to_horizon(
                date, earth_night_lights.points
            ).flatten()
            new_nl_opacity = (
                np.clip(ang_nl, -sunset_tol, sunset_tol) / sunset_tol
                + 1
            )
            new_nl_scalars = np.hstack(
                (
                    earth_night_lights.active_scalars[:, :3],
                    new_nl_opacity.reshape((-1, 1)),
                )
            )
            pl.update_scalars(new_nl_scalars, earth_night_lights)

        pl.actors["earth_night_lights"].user_matrix = rotm4_nl
        pl.actors["earth"].visibility = True
        pl.actors["earth_night_lights"].visibility = True




[docs]def scatter3(pl: pv.Plotter, v: np.ndarray, **kwargs) -> pv.PolyData:
    """Replicates MATLAB scatter3() with ``pyvista`` backend

    :param pl: Plotter object to add points to
    :type pl: pv.Plotter
    :param v: Vector to scatter
    :type v: np.ndarray [nx3]
    :param **kwargs: Additional arguments passed to ``pl.add_mesh``
    :return: Handle to plotted data
    :rtype: pv.PolyData
    """
    assert 3 in v.shape, TypeError(
        "scatter3 requires a 3xn or nx3 input vector"
    )
    if v.shape[0] == 3:
        v = np.transpose(v)

    pc = pv.PolyData(v)
    pl.add_mesh(pc, render_points_as_spheres=True, **kwargs)
    return pc




[docs]def plot3(pl: pv.Plotter, v: np.ndarray, **kwargs) -> pv.PolyData:
    """Replicates MATLAB plot3() with pyvista backend, please use densely scattered points to avoid confusing splines

    :param pl: Plotter object to add line to
    :type pl: pv.Plotter
    :param v: Vector to plot
    :type v: np.ndarray [nx3]
    :param **kwargs: Additional arguments passed to ``pl.add_mesh``
    :return: Plotted spline
    :rtype: pv.PolyData
    """
    assert 3 in v.shape, TypeError(
        "plot3 requires a 3xn or nx3 input vector"
    )
    if v.shape[0] == 3:
        v = np.transpose(v)

    spline = pv.Spline(v, v.shape[0])
    pl.add_mesh(spline, render_lines_as_tubes=True, **kwargs)
    return spline

    os.remove(imf)

[docs]def plot_angle_between(
    pl: pv.Plotter,
    v1: np.ndarray,
    v2: np.ndarray,
    center: np.ndarray,
    dist: float = 1.0,
    linewidth: float = 10.0,
):
    v1h, v2h = hat(v1), hat(v2)
    p1 = v1h * dist + center
    p2 = v2h * dist + center
    arc = pv.CircularArc(p1.flatten(), p2.flatten(), center.flatten())
    pl.add_mesh(arc, color="k", line_width=linewidth)
