[docs]def render_video(
    pre_render_fcn: Callable,
    render_fcn: Callable,
    post_render_fcn: Callable,
    frame_count: int,
    fname: str = None,
    framerate: int = 60,
    quality: int = 9,
    background_color: str = "white",
) -> None:
    """Renders a video given callables each stage of the recording

    :param pre_render_fcn: Called once before the first frame
    :type pre_render_fcn: Callable(pl: pv.Plotter)
    :param render_fcn: Called each frame
    :type render_fcn: Callable(pl: pv.Plotter, i: int)
    :param post_render_fcn: Called after each frame, before the next (only needed for niche effects)
    :type post_render_fcn: Callable(pl: pv.Plotter, i: int)
    :param frame_count: Number of frames to render
    :type frame_count: int
    :param fname: File name to save as, defaults to None
    :type fname: str, optional
    :param framerate: Frame rate of movie [frames/second], defaults to 60
    :type framerate: int, optional
    :param quality: Quality of movie file, defaults to 9, less than 10
    :type quality: int, optional
    :param background_color: Background color of plot window, defaults to "white"
    :type background_color: str, optional
    """
    pl = pv.Plotter()
    if fname is not None:
        if fname[-3:] == "mp4":
            pl.open_movie(fname, framerate=framerate, quality=quality)
        elif fname[-3:] == "gif":
            pl.open_gif(fname, fps=framerate)
    pl.set_background(background_color)
    pre_render_fcn(pl)
    pl._on_first_render_request()
    pl.render()
    for i in range(frame_count):
        render_fcn(pl, i)
        if fname is not None:
            pl.write_frame()
        else:
            pl.update()
        post_render_fcn(pl, i)
    pl.close()

    :type obj: SpaceObject

    :type quat: np.ndarray [nx4]

    """

        if fname is not None:

def _build_topo_interpolator() -> Callable:
    """Builds interpolators for elevation and bathymetry data from internal high-res image datasets

    :return: Interpolator that takes in latitude [rad] and longitude [rad] and returns elevation in km
    :rtype: Callable
    """
    max_elev_meters, min_bathy_meters = 6400, -8000
    with Image.open(
        os.path.join(os.environ["TEXDIR"], "earth_elev21600x10800.png")
    ) as im:
        im_elev = np.array(im, dtype=np.uint8)
        _LAT_SPACES = np.linspace(
            np.pi / 2, -np.pi / 2, im_elev.shape[0]
        )
        _LON_SPACES = np.linspace(-np.pi, np.pi, im_elev.shape[1])
        elev_interp = lambda lat, lon: scipy.interpolate.interpn(
            points=(
                _LAT_SPACES,
                _LON_SPACES,
            ),
            values=im_elev
            / 255.0
            * max_elev_meters,  # Scales image values to
            xi=(lat, lon),
            method="linear",
        )
    with Image.open(
        os.path.join(
            os.environ["TEXDIR"], "gebco_08_rev_bath_21600x10800.png"
        )
    ) as im:
        im_bathy = np.array(im, dtype=np.uint8)
        _LAT_SPACES = np.linspace(
            np.pi / 2, -np.pi / 2, im_bathy.shape[0]
        )
        _LON_SPACES = np.linspace(-np.pi, np.pi, im_bathy.shape[1])
        bathy_interp = lambda lat, lon: scipy.interpolate.interpn(
            points=(
                _LAT_SPACES,
                _LON_SPACES,
            ),
            values=(1 - im_bathy / 255.0)
            * min_bathy_meters,  # Scales image values to
            xi=(lat, lon),
            method="linear",
        )

    return (
        lambda lat, lon: bathy_interp(lat, lon) / 1e3
        + elev_interp(lat, lon) / 1e3
    )  # /1e3 to output meters


def _textured_ellipsoid(
    a: float, f: float, rotm: np.ndarray = np.eye(3)
) -> pv.ParametricEllipsoid:
    """Returns a parametric ellipsoid based on the given semimajor axis and flattening ratio, ready for texturing

    :param a: Semimajor axis of the body
    :type a: float
    :param f: Flattening ratio of the body
    :type f: float
    :param rotm: Rotation matrix to apply to the body points, defaults to np.eye(3)
    :type rotm: np.ndarray [3x3]
    :return: Ellipsoid parametrized by ``a``, ``f``
    :rtype: pv.ParametricEllipsoid
    """
    b = a * (1 - f)  # Semi-minor axis
    body = pv.ParametricEllipsoid(
        a,
        a,
        b,
        u_res=_PLANET_RES,
        v_res=_PLANET_RES,
        w_res=_PLANET_RES,
        min_u=-np.pi / 2 + 0.00001,
        max_u=3 * np.pi / 2 - 0.00001,
    )
    sph_hat_pts = hat(body.points)
    body.active_t_coords = np.zeros((body.points.shape[0], 2))
    body.active_t_coords[:, 0] = 0.5 + np.arctan2(
        -sph_hat_pts[:, 0], sph_hat_pts[:, 1]
    ) / (2 * np.pi)
    body.active_t_coords[:, 1] = (
        0.5 + np.arcsin(sph_hat_pts[:, 2]) / np.pi
    )
    body.points = (rotm @ body.points.T).T
    return body

    pl: pv.Plotter,
    mode: str = "eci",
    date: datetime.datetime = None,
    high_def: bool = False,
    stars: bool = False,
    night_lights: bool = False,
    elevation: bool = False,
    use_elevation_texture: bool = False,
    atmosphere: bool = False,
    lighting: bool = True,
    borders: bool = False,
) -> pv.PolyData:
    """Plots the Earth as a sphere in [km]

    :param pl: Plotter to use
    :type pl: pv.Plotter
    :return: Generated spherical mesh
    :rtype: pv.PolyData
    """
    if date is None:
        date = now()
    pl.enable_terrain_style(True, True)
    pl.enable_anti_aliasing("ssaa", all_renderers=False)
    if elevation:
        interp_elev = _build_topo_interpolator()
    global earth_night_lights, earth_atmosphere
    sunset_tol = np.deg2rad(1)
    assert mode in ["eci", "ecef"], "Mode must be 'eci' or 'ecef'"
    (_, _, r3) = axis_rotation_matrices()
    rotm4 = np.eye(4)
    if mode == "eci":
        rotm4[:3, :3] = ecef_to_eci(date)

    if "earth" not in pl.actors:
        if lighting:
            pl.remove_all_lights(only_active=True)
            pl.add_light(_SUNLIGHT, only_active=True)

        if atmosphere:
            earth_atmosphere = _textured_ellipsoid(
                AstroConstants.earth_r_eq + 100,
                AstroConstants.earth_f,
                r3(np.pi / 2),
            )
            pl.add_mesh(
                earth_atmosphere,
                opacity=0 * earth_atmosphere.points[:, 0],
                name="atmosphere",
                show_scalar_bar=False,
                texture=pv.Texture(
                    os.path.join(
                        os.environ["TEXDIR"],
                        "storm_clouds_8k_edited.png",
                    )
                ),  # http://shadedrelief.com/natural3/ne3_data/8192/clouds/storm_clouds_8k.jpg
                specular_power=128,
                specular=1,
                diffuse=1,
                smooth_shading=True,
            )
            pl.actors["atmosphere"].prop.culling = "front"

        earth = _textured_ellipsoid(
            AstroConstants.earth_r_eq,
            AstroConstants.earth_f,
            r3(np.pi / 2),
        )

        if borders:
            sf = shapefile.Reader(
                os.path.join(
                    os.environ["SRCDIR"],
                    "resources",
                    "shp",
                    "World_Countries_Generalized.shp",
                )
            )
            shapes = sf.shapes()
            plot_shape = lambda r_itrf: plot3(
                pl,
                np.vstack((r_itrf, r_itrf[[0], :])),
                lighting=False,
                line_width=3,
                color="y",
                opacity=0.5,
                name="borders",
            )
            shape_itrf = np.empty((0, 3))
            for shape in shapes:
                shape_lla = np.deg2rad(np.vstack(shape.points))
                if len(shape.parts) == 1:
                    part_itrf = lla_to_itrf(
                        shape_lla[:, 1],
                        shape_lla[:, 0],
                        0 * shape_lla[:, 0],
                    )
                    shape_itrf = np.vstack(
                        (
                            shape_itrf,
                            [[np.nan, np.nan, np.nan]],
                            part_itrf,
                            part_itrf[[0], :],
                        )
                    )
                else:
                    for i in range(len(shape.parts)):
                        sind = shape.parts[i]
                        eind = (
                            shape.parts[i + 1]
                            if i + 1 < len(shape.parts)
                            else len(shape.points) - 1
                        )
                        if eind - sind > 100:
                            part_itrf = lla_to_itrf(
                                shape_lla[sind:eind, 1],
                                shape_lla[sind:eind, 0],
                                0 * shape_lla[sind:eind, 0],
                            )
                            shape_itrf = np.vstack(
                                (
                                    shape_itrf,
                                    [[np.nan, np.nan, np.nan]],
                                    part_itrf,
                                    part_itrf[[0], :],
                                )
                            )

            plot_shape(shape_itrf)

        if elevation:
            (lat_pts, lon_pts, _) = itrf_to_lla(earth.points)
            elev_at_pts = interp_elev(lat_pts, lon_pts)
            elev_at_pts[0] = np.abs(np.min(elev_at_pts))
            earth["elev_vector"] = (
                20 * elev_at_pts.reshape((-1, 1)) * hat(earth.points)
            )
            earth["elev_scalar"] = elev_at_pts
            earth.warp_by_vector("elev_vector", inplace=True)

        if high_def:
            tex_path = os.path.join(
                os.environ["TEXDIR"],
                "blue_marble.jpg",
            )
        else:
            tex_path = os.path.join(
                os.environ["TEXDIR"],
                "earth_tex.jpg",
            )
        etex = pv.Texture(tex_path)
        pl.add_mesh(
            earth,
            texture=etex if not use_elevation_texture else None,
            cmap="topo",
            smooth_shading=True,
            name="earth",
            scalars="elev_scalar" if use_elevation_texture else None,
            diffuse=0.7,
            specular=0.6,
            specular_power=100,
        )
        pl.set_background("black")  # Because obviously, space and stuff
        if stars:
            cubemap = pv.cubemap(
                os.path.join(os.environ["TEXDIR"], "16k"), ext=".png"
            )
            pl.add_actor(cubemap.to_skybox())

        if night_lights:
            pts = np.fromfile(
                os.path.join(
                    os.environ["TEXDIR"],
                    "night_lights_ecef_tol_255.pkl",
                )
            ).reshape((-1, 4))
            ang_nl = -sun_angle_to_horizon(date, pts[:, :3]).flatten()
            pts[:, 3] = (
                np.clip(ang_nl, -sunset_tol, sunset_tol) / sunset_tol
                + 1
            )

            earth_night_lights = pv.PolyData(pts[:, :3])
            pl.add_points(
                earth_night_lights,
                style="points_gaussian",
                color="#ffffe6",
                point_size=0.5,
                opacity=pts[:, 3],
                name="earth_night_lights",
            )

    # Every time:
    pl.actors["earth"].user_matrix = rotm4
    if lighting:
        _update_sunlight(pl, date, mode)

    if borders:
        pl.actors["borders"].user_matrix = rotm4

    if atmosphere:
        ang = sun_angle_to_horizon(
            date, earth_atmosphere.points
        ).flatten()
        new_scalars = earth_atmosphere.active_scalars
        new_scalars[:, -1] = 255 * np.clip(ang - sunset_tol, 0, 1)
        # new_scalars[:, -1] = 255
        pl.update_scalars(new_scalars, earth_atmosphere)
        pl.actors["atmosphere"].user_matrix = rotm4

        ).flatten()
        new_scalars = earth_night_lights.active_scalars
        new_scalars[:, -1] = (
            np.clip(ang_nl, -sunset_tol, sunset_tol) / sunset_tol + 1
        )

        pl.update_scalars(new_scalars, earth_night_lights)

        pl.actors["earth_night_lights"].user_matrix = rotm4




[docs]def plot_moon(
    pl: pv.Plotter,
    mode: str = "eci",
    date: datetime.datetime = None,
    lighting: bool = True,
) -> pv.PolyData:
    """Plots the Moon as a sphere in [km]

    :param pl: Plotter to use
    :type pl: pv.Plotter
    :return: Generated spherical mesh
    :rtype: pv.PolyData
    """
    assert mode in [
        "eci",
        "ecef",
        "mci",
    ], "Mode must be 'eci', 'ecef', or 'mci'"

    transm4 = np.eye(4)
    rotm4orb = np.eye(4)
    if mode == "eci":
        transm4[:3, -1] = moon(date)
        rotm4orb[:3, :3] = mcmf_to_j2000(date)
    elif mode == "ecef":
        transm4[:3, -1] = (eci_to_ecef(date) @ moon(date).T).T
        rotm4orb[:3, :3] = eci_to_ecef(date) @ mcmf_to_j2000(date)
    elif mode == "emi":
        transm4[:3, -1] = np.array([0.0, 0.0, 0.0])

    if "moon" not in pl.actors:
        if lighting:
            pl.remove_all_lights(only_active=True)
            pl.add_light(_SUNLIGHT, only_active=True)
        pl.enable_terrain_style(True, True)
        pl.enable_anti_aliasing("ssaa", all_renderers=False)

        moon_mesh = _textured_ellipsoid(
            AstroConstants.moon_r_eq, AstroConstants.moon_f
        )
        moon_mesh.rotate_z(
            np.rad2deg(-np.pi / 2), point=(0.0, 0.0, 0.0), inplace=True
        )

        tex_path = os.path.join(
            os.environ["TEXDIR"],
            "moon_tex.jpg",
        )
        tex = pv.Texture(tex_path)
        pl.add_mesh(
            moon_mesh,
            texture=tex,
            cmap="topo",
            smooth_shading=True,
            name="moon",
        )
        pl.set_background("black")  # Because obviously, space and stuff
    pl.actors["moon"].user_matrix = transm4 @ rotm4orb

    if lighting:
        _update_sunlight(pl, date, mode)



def _update_sunlight(
    pl: pv.Plotter, date: datetime.datetime, mode: str
) -> None:
    # Assumes that there is only one light in the renderer
    light_coll = pl.renderer.GetLights()
    light_coll.InitTraversal()
    sun_light = light_coll.GetNextItem()
    if mode == "eci" or mode == "mci":
        sun_light.SetPosition(*sun(date).flatten())
    elif mode == "ecef":
        sun_light.SetPosition(
            *(eci_to_ecef(date) @ sun(date).T).flatten()
        )



[docs]def scatter3(pl: pv.Plotter, v: np.ndarray, **kwargs) -> pv.PolyData:
    """Replicates MATLAB's ``scatter3()`` with ``pyvista`` backend

    :param pl: Plotter object to add points to
    :type pl: pv.Plotter
    :param v: Vector to scatter
    :type v: np.ndarray [nx3]
    :param **kwargs: Additional arguments passed to ``pl.add_mesh()``
    :return: Handle to plotted data
    :rtype: pv.PolyData
    """
    assert 3 in v.shape, TypeError(
        "scatter3 requires a 3xn or nx3 input vector"
    )
    if v.shape[0] == 3:
        v = np.transpose(v)

    pc = pv.PolyData(v)
    pl.add_mesh(pc, render_points_as_spheres=True, **kwargs)
    pl.camera.focal_point = (0.0, 0.0, 0.0)
    return pc




[docs]def plot3(
    pl: pv.Plotter, v: np.ndarray, use_spline: bool = False, **kwargs
) -> pv.PolyData:
    """Replicates MATLAB's ``plot3()`` with a ``pyvista`` backend, please use densely scattered points to avoid confusing splines

    :param pl: Plotter object to add line to
    :type pl: pv.Plotter
    :param v: Vector to plot
    :type v: np.ndarray [nx3]
    :param use_spline: Whether to plot as a spline, defaults to False
    :type use_spline: bool
    :param **kwargs: Additional arguments passed to ``pl.add_mesh()``
    :return: Plotted spline
    :rtype: pv.PolyData
    """
    assert 3 in v.shape, TypeError(
        "plot3 requires a 3xn or nx3 input vector"
    )
    if v.shape[0] == 3:
        v = np.transpose(v)

    spline = (
        pv.Spline(v, v.shape[0]) if use_spline else pv.MultipleLines(v)
    )
    pl.add_mesh(spline, render_lines_as_tubes=True, **kwargs)
    pl.camera.focal_point = (0.0, 0.0, 0.0)
    return spline

    pl: pv.Plotter,
    v1: np.ndarray,
    v2: np.ndarray,

    dist: float = 1.0,

        "shape_opacity": shape_opacity,
    }

        **plot_kwargs,

        name=name,

            origin + direction * scale,
            [label],
            **label_kwargs,
