[docs]def eci_to_sun_ec(dates: np.ndarray[datetime.datetime]) -> np.ndarray:
    n = dates.size
    v1 = hat(sun(date_to_jd(dates)))
    ref = np.tile([0, 0, 1], (n, 1))
    v2 = hat(np.cross(v1, ref))
    v3 = np.cross(v1, v2)
    dcms = np.dstack((v1, v2, v3))
    return dcms

[docs]def ecef_to_eci(date) -> np.ndarray:
    """Converts from Earth-centered, Earth-fixed (ECEF) coordinates to Earth- centered inertial (ECI). Does not account for precession, nutation, or polar motion.

    :param date: Time
    :type date: datetime.datetime [UTC]
    :return: Rotation matrix from ECEF to ECI
    :rtype: np.ndarray [3x3]
    """
    (_, _, r3) = axis_rotation_matrices()
    sid_time = date_to_sidereal(date)
    # Gets the current sidereal time
    gmst = (sid_time % 86400) / 86400 * 2 * np.pi
    # Gets the GMST hour angle
    return r3(
        -gmst
    )  # Gets the rotation matrix about the third body axis




[docs]def moon(jd: np.ndarray) -> np.ndarray:
    """Translation of Vallado's MATLAB method of the same name, approximates the position of the Moon.

    :param jd: Julian dates
    :type jd: np.ndarray [nx1]
    :return: Approximate vector to Moon from Earth center [km]
    :rtype: np.ndarray [nx3]
    """
    jd = np.array([jd]).flatten()
    ttdb = (jd - 2451545.0) / 36525.0

    eclplong = wrap_to_360(
        218.32
        + 481267.8813 * ttdb
        + 6.29 * sind(134.9 + 477198.85 * ttdb)
        - 1.27 * sind(259.2 - 413335.38 * ttdb)
        + 0.66 * sind(235.7 + 890534.23 * ttdb)
        + 0.21 * sind(269.9 + 954397.70 * ttdb)
        - 0.19 * sind(357.5 + 35999.05 * ttdb)
        - 0.11 * sind(186.6 + 966404.05 * ttdb)
    )

    eclplat = wrap_to_360(
        5.13 * sind(93.3 + 483202.03 * ttdb)
        + 0.28 * sind(228.2 + 960400.87 * ttdb)
        - 0.28 * sind(318.3 + 6003.18 * ttdb)
        - 0.17 * sind(217.6 - 407332.20 * ttdb)
    )

    hzparal = wrap_to_360(
        0.9508
        + 0.0518 * cosd(134.9 + 477198.85 * ttdb)
        + 0.0095 * cosd(259.2 - 413335.38 * ttdb)
        + 0.0078 * cosd(235.7 + 890534.23 * ttdb)
        + 0.0028 * cosd(269.9 + 954397.70 * ttdb)
    )

    obliquity = 23.439291 - 0.0130042 * ttdb
    magr = 1.0 / sind(hzparal)
    # ------------- calculate moon position vector ----------------
    rmoon = np.empty((jd.shape[0], 3), dtype=np.float64)
    rmoon[:, 0] = magr * cosd(eclplat) * cosd(eclplong)
    rmoon[:, 1] = magr * (
        cosd(obliquity) * cosd(eclplat) * sind(eclplong)
        - sind(obliquity) * sind(eclplat)
    )
    rmoon[:, 2] = magr * (
        sind(obliquity) * cosd(eclplat) * sind(eclplong)
        + cosd(obliquity) * sind(eclplat)
    )
    return rmoon * AstroConstants.earth_r_eq




[docs]def sun(jd: np.ndarray) -> np.ndarray:
    """Translation of Vallado's MATLAB method of the same name, approximating the position of the sun.

    :param jd: Julian days
    :type jd: np.ndarray [nx1]
    :return: Approximate vector to the Sun from Earth center [km]
    :rtype: np.ndarray [nx3]
    """
    jd = np.array([jd]).flatten()
    tut1 = (jd - 2451545.0) / 36525.0
    meanlong = (280.460 + 36000.77 * tut1) % 360.0
    meananomaly = np.deg2rad((357.5277233 + 35999.05034 * tut1) % 360.0)
    meananomaly = np.array(
        [m if m > 0 else m + 2 * np.pi for m in meananomaly]
    )
    # if meananomaly < 0: meananomaly += 2*np.pi
    eclplong = np.deg2rad(
        (
            meanlong
            + 1.914666471 * np.sin(meananomaly)
            + 0.019994643 * np.sin(2 * meananomaly)
        )
        % 360.0
    )
    obliquity = np.deg2rad(23.439291 - 0.0130042 * tut1)
    magr = (
        1.000140612
        - 0.016708617 * np.cos(meananomaly)
        - 0.000139589 * np.cos(2.0 * meananomaly)
    )
    # in AU
    rsun = np.empty((jd.shape[0], 3))
    rsun[:, 0] = magr * np.cos(eclplong)
    rsun[:, 1] = magr * np.cos(obliquity) * np.sin(eclplong)
    rsun[:, 2] = magr * np.sin(obliquity) * np.sin(eclplong)
    return rsun * AstroConstants.au_to_km




[docs]def moon_brightness(
    alpha: np.ndarray,
    rho: np.ndarray,
    k: np.ndarray,
    zm: np.ndarray,
    z: np.ndarray,
) -> np.ndarray:
    """Calculates brightness of the Moon following `source <https://iopscience.iop.org/article/10.1086/132921/pdf?casa_token=gZtuzfmx-fgAAAAA:D_WKhr-5oTRMBLQyDy5oeBvgeBc9QAd4lHvJKf7gFney_lw6KxzYjf5c7c7h0TYRqkbL7pQR16Ct8jghpbE>`_

    :param alpha: Lunar phase angle, angular distance between Earth and Sun as seen from Moon [rad]
    :type alpha: np.ndarray [nx1]
    :param rho: Moon/observation separation angle [rad]
    :type rho: np.ndarray [nx1]
    :param k: Extinction coefficient [mag/airmass]
    :type k: np.ndarray [nx1]
    :param zm: Angle of the moon from local zenith
    :type zm: np.ndarray [nx1]
    :param z: Angle of the observation from local zenith
    :type z: np.ndarray [nx1]
    :return: Brightness of the moon in given geometries [mag/arcsecond^2]
    :rtype: np.ndarray [nx1]
    """
    alpha = np.rad2deg(alpha)
    istar = 10 ** (
        -0.4 * (3.84 + 0.026 * np.abs(alpha) + 4e-9 * alpha**4)
    )
    frho = 10**5.36 * (1.06 + np.cos(rho) ** 2) + 10 ** (
        6.15 - np.rad2deg(rho) / 40
    )
    x = lambda y: (1 - 0.96 * np.sin(y) ** 2) ** (-0.5)
    bmoon = (
        frho
        * istar
        * 10 ** (-0.4 * k * x(zm))
        * (1 - 10 ** (-0.4 * k * x(z)))
    )
    vmoon_per_sqasec = (20.7233 - np.log(bmoon / 34.08)) / 0.92104
    return bmoon, vmoon_per_sqasec




[docs]def twilight_brightness(
    z: np.ndarray, delta_az: np.ndarray, h: np.ndarray
) -> np.ndarray:
    """Calculates brightness magnitude of twilight sky.

    :param z: Angle of observation from zenith [rad]
    :type z: np.ndarray
    :param delta_az: Angle between observation and Sun, on the local horizon [rad]
    :type delta_az: np.ndarray
    :param h: Elevation of the Sun, ``h<0`` if below horizon [rad]
    :type h: np.ndarray
    :return: Brightness magnitude due to the setting Sun [mag/arcsec^2]
    :rtype: np.ndarray
    """
    z, delta_az, h = (
        np.rad2deg(z),
        np.rad2deg(delta_az),
        np.rad2deg(h),
    )
    m_twilight = (
        -(7.5e-5 * z + 5.05e-3) * delta_az
        + (3.67e-4 * z - 0.458) * h
        + 9.17e-3 * z
        + 3.535
    )
    transition_angle = -(4.12e-2 * z + 0.582) * h + 0.417 * z + 97.5
    m_twilight2 = (
        -0.0010 * delta_az
        + (1.12e-3 * z - 0.470) * h
        - 4.17e-3 * z
        + 3.225
    )
    m_twilight[delta_az > transition_angle] = m_twilight2[
        delta_az > transition_angle
    ]

    m_twilight[np.argwhere(h > 0)] = np.nan  # Daylight!
    m_twilight[np.argwhere(h < -18)] = np.nan  # Night!
    return m_twilight




[docs]def twilight_brightness_correction(
    z: np.ndarray, delta_az: np.ndarray, h: np.ndarray
) -> np.ndarray:
    m90 = twilight_brightness(z=np.pi / 2, delta_az=delta_az, h=h)
    m30 = twilight_brightness(z=np.pi / 6, delta_az=delta_az, h=h)
    z = np.rad2deg(z)
    return (z - 60) / 30 * m90 + (90 - z) / 30 * m30




[docs]def light_pollution_brightness(zlp: float, z: np.ndarray) -> np.ndarray:
    """Computes magnitude of sky brightness due to light pollution.

    :param zlp: Zenith level of light pollution [mag/arcsec^2]
    :type zlp: float
    :param z: Angle of observation from zenith [rad]
    :type z: np.ndarray
    :return: Light pollution level [mag/arcsec^2]
    :rtype: np.ndarray
    """
    a = 319.186511755673
    b = -46.9419919726453
    c = 0.434369283318305
    d = 2.39138122541257
    e = -0.00121311966265414
    f = -0.0488735671014705
    g = -0.0396262989070584
    h = -9.28758741258742e-7
    i = 0.0000548215089277455
    j = 0.00134667559232511

    z = np.rad2deg(z)

    mlp = (
        a
        + b * zlp
        + c * z
        + d * zlp**2
        + e * z**2
        + f * zlp * z
        + g * zlp**3
        + h * z**3
        + i * zlp * z**2
        + j * zlp**2 * z
    )
    return mlp
