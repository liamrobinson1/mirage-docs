    def __init__(self, preset: str = "pogs", constraints: dict = None):
        if preset.lower() == "pogs":

            )

        self.lambdas = []

                self.lambdas.append(
                    self.set_constraint_lambda(key, val)
                )


[docs]    def set_constraint_lambda(self, key, val) -> Callable:
        if key == "min_elevation":
            return lambda **kwargs: is_elevation(
                **kwargs, min_elevation_rad=np.deg2rad(val)
            )
        elif key == "obs_eclipsed" and val:
            return lambda **kwargs: is_eclipsed_geodetic_surface(
                lat_geod=self.lat_geod_rad,
                lon_rad=self.lon_rad,
                **kwargs,
            )
        elif key == "target_illuminated" and val:
            return lambda **kwargs: np.array(
                [not b for b in is_eclipsed_geocentric_eci(**kwargs)]
            )
        elif key == "moon_exclusion":
            return lambda **kwargs: is_moon_exclusion(
                **kwargs, moon_exclusion_angle_min_rad=np.deg2rad(val)
            )
        elif key == "visual_mag_limit":
            return lambda **kwargs: is_visual_magnitude(
                **kwargs, maximum_visual_magnitude=val
            )
        else:
            raise KeyError(
                f"Observation constraint '{val}' not implemented!"
            )

[docs]    def eval_constraints(self, **kwargs) -> np.ndarray[bool]:
        constraint_results = np.array(
            [lam(**kwargs).flatten() for lam in self.lambdas]
        ).T
        all_constraints_satisfied = np.all(constraint_results, axis=1)
        return all_constraints_satisfied



[docs]    def moon_brightness_at_dates(
        self,
        dates: np.ndarray[datetime.datetime],
        look_dir_eci: np.ndarray,
    ) -> np.ndarray:
        station_eci = self.eci_at_dates(dates)
        moon_eci = moon(date_to_jd(dates))
        sun_eci = sun(date_to_jd(dates))
        station_to_moon = moon_eci - station_eci
        station_to_sun = sun_eci - station_eci
        moon_to_sun = station_to_sun - station_to_moon
        lunar_phase_angle = np.arccos(
            dot(hat(moon_to_sun), -hat(station_to_moon))
        )
        moon_angle_from_zenith = np.arccos(
            dot(hat(station_eci), hat(moon_eci))
        )
        look_angle_from_zenith = np.arccos(
            dot(hat(station_eci), hat(look_dir_eci))
        )
        moon_obs_angle = np.arccos(
            dot(hat(look_dir_eci), hat(station_to_moon))
        )
        return moon_brightness(
            alpha=lunar_phase_angle,
            rho=moon_obs_angle,
            k=0.116,
            zm=moon_angle_from_zenith,
            z=look_angle_from_zenith,
        )

    def _sky_pixel_irradiance(
        self,
        dates: np.ndarray[datetime.datetime],
        look_dir_eci: np.ndarray,
    ) -> np.ndarray:
        m_b_per_asec2 = self._total_sky_brightness(dates, look_dir_eci)
        i_b_per_asec2 = apparent_magnitude_to_irradiance(m_b_per_asec2)
        i_b_per_pix = i_b_per_asec2 * self.telescope.pixel_scale**2
        return i_b_per_pix  # Expected irradiance per pixel


[docs]    def sky_noise(
        self,
        dates: np.ndarray[datetime.datetime],
        look_dir_eci: np.ndarray,
        i_obj: np.ndarray,
    ) -> np.ndarray:
        i_b_pix = self._sky_pixel_irradiance(dates, look_dir_eci)
        signal_pix = 50
        i_b_under = i_b_pix * signal_pix
        snr = 10 * np.log10(i_obj / i_b_under)
        noise_background = np.random.normal(
            loc=i_b_under,
            scale=np.sqrt(i_b_under),
            size=(dates.size, 1),
        )
        return noise_background, snr, i_b_under

[docs]def scenario_light_curve(
    obj: SpaceObject,
    obj_attitude: RbtfAttitude,
    station: Station,
    brdf: Brdf,
    date_space: np.ndarray[datetime.datetime],
    noise_def: dict,
) -> np.ndarray:
    epsec_space = np.array(
        [(d - date_space[0]).total_seconds() for d in date_space]
    )
    (q, w) = obj_attitude.propagate(epsec_space)
    (r_sat, _) = obj.propagate(date_space)
    r_station = station.eci_at_dates(date_space)

    sat_to_sun_inertial = sun(date_to_jd(date_space))
    sat_to_station_inertial = r_station - r_sat
    rmag_station_to_sat = vecnorm(sat_to_station_inertial)

    sat_to_sun_body = stack_mat_mult(
        quat_to_dcm(q), sat_to_sun_inertial
    )
    sat_to_obs_body = stack_mat_mult(
        quat_to_dcm(q), sat_to_station_inertial
    )

    # Generating noise
    noise = np.zeros((date_space.size, 1))
    if noise_def["type"].lower() == "gaussian":
        noise = np.random.normal(
            loc=0, scale=noise_def["std"], size=(date_space.size, 1)
        )
    elif noise_def["type"].lower() == "poisson":
        noise = np.random.poisson(
            lam=noise_def["std"], size=(date_space.size, 1)
        )
    else:
        raise NotImplementedError("Noise type not implemented")

    lc_normalized = run_engine(
        brdf, obj.file_name, hat(sat_to_sun_body), hat(sat_to_obs_body)
    )

    lc = lc_normalized / rmag_station_to_sat**2

    constraint_results = station.eval_constraints(
        dates=date_space,
        target_pos_eci=r_sat,
        obs_pos_eci=r_station,
        look_dir_eci=-sat_to_station_inertial,
        lc=lc,
    )

    lc[np.logical_not(constraint_results).flatten()] = np.nan
    return (
        lc,
        {
            "constraint_satisfaction": constraint_results,
            "object_quat": q,
            "object_omega": w,
            "rmag_station_to_sat": rmag_station_to_sat,
            "object_pos_eci": r_sat,
            "station_pos_eci": r_station,
            "lc_normalized": lc_normalized,
        },
    )
