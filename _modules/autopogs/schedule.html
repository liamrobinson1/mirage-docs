class SingleTelescopeReservation(TelescopeReservation):
    def __init__(
        self,
        start_time: datetime.datetime,
        end_time: datetime.datetime,
        target: int,
        t_int: float,
        n_images: int,
        user_email: str,
        priority: int = 0,
    ):
        """A single telescope reservation

        :param start_time: The time after which the reservation is valid
        :type start_time: datetime.datetime
        :param end_time: The final time the reservation is valid for
        :type end_time: datetime.datetime
        :param target: The NORAD ID of the target
        :type target: int
        :param t_int: The integration time for each image
        :type t_int: float
        :param n_images: The number of images to take
        :type n_images: int
        :param user_email: The email of the user making the reservation
        :type user_email: str
        :param priority: The priority of the reservation, defaults to 0
        :type priority: int, optional
        """
        self.start_time = start_time
        self.end_time = end_time
        self.target = target
        self.t_int = t_int
        self.n_images = n_images
        self.user_email = user_email
        self.priority = priority

        self.reservation_function = self._reserve
        self.validate()

        self.desc_dict = json.dumps(
            {
                "priority": self.priority,
                "target": self.target,
                "t_int": self.t_int,
                "n_images": self.n_images,
            }
        )

    def _reserve(self):
        """Builds the schedule for a single reservation"""
        attendees = [{"email": self.user_email, "responseStatus": "accepted"}]
        reminders = {
            "useDefault": False,
            "overrides": [
                {"method": "email", "minutes": 0},
            ],
        }
        create_calendar_event(
            calendar_id=os.environ["POGS_CALENDAR_ID"],
            event_name=f"Observing {self.target}",
            event_description=self.desc_dict,
            event_time_start=self.start_time,
            event_time_end=self.end_time,
            attendees=attendees,
            # reminders=reminders
        )
