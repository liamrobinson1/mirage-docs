    import os

    import mirage as mr


    def count_lines_in_dir(rel_dir):
        lines = 0
        for fname in os.listdir(rel_dir):
            if ".py" in fname:
                with open(os.path.join(rel_dir, fname), "r") as f:
                    lines += len(f.readlines())
        return lines


    rel_dirs = [

    ]

    lines = 0

        dir_ = os.path.join(os.environ["SRCDIR"], rel_dir)
        lines += count_lines_in_dir(dir_)
