    import numpy as np

    import mirage as mr
    import mirage.vis as mrv

    station = mr.Station(preset='pogs')
    t2 = mr.load_json_data('tycho2.json')
    alpha_rad, delta_rad = t2['j2000_ra'], t2['j2000_dec']
    mr.tic()
    alpha_rad_app, delta_rad_app = mr.apparent_star_positons(mr.now(), 
                                                             station.lat_geod_rad, 
                                                             alpha_rad, 
                                                             delta_rad)
    mr.toc()
